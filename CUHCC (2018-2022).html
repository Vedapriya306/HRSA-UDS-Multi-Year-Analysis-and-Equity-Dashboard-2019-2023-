Cardiovascular <- CUHCC_Clinical_data %>%

    Year = `Year`,

# To establish the mini and maximum values and text on the plot we create a dataframe  
#Calculate min and max values for each category
min_max_df <- Cuhcc_Cardio %>%
  group_by(CARDIOVASCULAR) %>%
  summarise(
    Min_Value = min(Value),
    Max_Value = max(Value),
    Min_Year = Year[which.min(Value)],
    Max_Year = Year[which.max(Value)]
  )
# Create the plot using ggplot2
# plot() funciton is used to have significt grid lines, background and axis
#guide_legend is used for customize the legend appearance and title.

plot <- ggplot(Cuhcc_Cardio, aes(x = Year, y = Value, color = CARDIOVASCULAR, group = CARDIOVASCULAR)) +
  geom_line(size = 1) +                      # Add lines for each category
  geom_point(size = 2) +                     # Add points for each data point
  theme_minimal() +                         # Use a minimal theme
  labs(title = "Trends in Cardiovascular Disease Over Time",
       x = "Year",
       y = "Percentage of Cardiovascular Categories") +
  scale_color_manual(values = c("Hypertension" = "blue", 
                                "Heart_Attack_Stroke_Treatment" = "red",
                                "Blood_Pressure_Control" = "purple",
                                "Statin_Therapy" = "green",
                                "Min Value" = "yellow",
                                "Max Value" = "orange")) +
  scale_y_continuous(labels = scales::percent_format(accuracy = 1), 
                     limits = c(0, 1),
                     breaks = seq(0, 1, by = 0.10)) +
  # Add points for min values
  geom_point(data = min_max_df, aes(x = Min_Year, y = Min_Value, color = "Min Value"), 
             size = 3, shape = 21, fill = "yellow") +
  # Add points for max values
  geom_point(data = min_max_df, aes(x = Max_Year, y = Max_Value, color = "Max Value"), 
             size = 3, shape = 21, fill = "orange") +
  # Add text labels for min values
  geom_text(data = min_max_df, aes(x = Min_Year, y = Min_Value, 
                                    label = paste("Min:", scales::percent(Min_Value))), 
            vjust = -1, color = "black") +
  # Add text labels for max values
  geom_text(data = min_max_df, aes(x = Max_Year, y = Max_Value,
                                    label = paste("Max:", scales::percent(Max_Value))), 
            vjust = -1, color = "black") +
  guides(color = guide_legend(title = "CARDIOVASCULAR TYPES",
                              title.position = "top",
                              title.hjust = 1,
                              override.aes = list(size = 3))) +
  theme(legend.position = "Bottom")
